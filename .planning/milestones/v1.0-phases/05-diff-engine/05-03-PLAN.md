```python
class DiffResult:
    added_nodes: tuple[AlteryxNode, ...]
    removed_nodes: tuple[AlteryxNode, ...]
    modified_nodes: tuple[NodeDiff, ...]
    edge_diffs: tuple[EdgeDiff, ...]
    is_empty: bool  # @property — True when all tuples empty

class NodeDiff:
    tool_id: ToolID
    old_node: AlteryxNode
    new_node: AlteryxNode
    field_diffs: dict[str, tuple[Any, Any]]  # "dotted.path" -> (before, after)

class EdgeDiff:
    src_tool: ToolID
    src_anchor: AnchorName
    dst_tool: ToolID
    dst_anchor: AnchorName
    change_type: str  # "added" or "removed"
```
